関数の実行時間を計測する

import time


def timeit(func):
    def w(*a, **kw):
	s = time.time()
	func(*a, *kw)
	e = (time.time() - s) * 1000
	print(f"{e:0.3} ms")

    return w

使い方

@timeit
def run():
    sum(i for i in range(1_000_000))

run()

30.9 ms
