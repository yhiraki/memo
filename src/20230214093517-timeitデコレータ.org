#+title: timeitデコレータ
* 関数の実行時間を計測する

#+name: timeit
#+begin_src python
  import time


  def timeit(func):
      def w(*a, **kw):
          s = time.time()
          func(*a, *kw)
          e = (time.time() - s) * 1000
          print(f"{e:0.3} ms")

      return w
#+end_src

* 使い方

#+begin_src python :noweb strip-export :exports both
  <<timeit>>
  @timeit
  def run():
      sum(i for i in range(1_000_000))

  run()
#+end_src
